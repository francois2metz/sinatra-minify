def initialize(type, file, package)
  @type        = type.to_s
  @file        = file
  @package     = package
  @command     = :minify_#{@type}"
  @host        = ENV["MINIFY_SITE_URL"] || "http://localhost:4567"

  raise ArgumentError  if not ['js', 'css'].include?(type)
end
